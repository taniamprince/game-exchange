CREATE TABLE CUSTOMER_ORDER(ID INT PRIMARY KEY AUTO_INCREMENT, CUSTOMER_NAME varchar (255), CUSTOMER_EMAIL varchar(255), STATUS varchar(255));

CREATE TABLE PAYMENT_INFO(ID INT PRIMARY KEY AUTO_INCREMENT, CUSTOMER_ORDER_ID_FK int, CARDHOLDER_NAME varchar (255), CC_NUM int, EXP varchar(255), CCV int, CONFORMATION_NUMBER int, INVALID varchar(255));

CREATE TABLE SHIPPING_INFO(ID INT PRIMARY KEY AUTO_INCREMENT, CUSTOMER_ORDER_ID_FK int, ADDRESS1 varchar(255), ADDRESS2 varchar(255), CITY varchar(255), STATE varchar(255), ZIP int, INVALID varchar(255));

CREATE TABLE ITEM(ID INT PRIMARY KEY AUTO_INCREMENT, NAME varchar(255), DESCRIPTION varchar(255), AVAILABLE_QUANTITY int, UNIT_PRICE decimal);

CREATE TABLE CUSTOMER_ORDER_LINE_ITEM(ID INT PRIMARY KEY AUTO_INCREMENT, CUSTOMER_ORDER_ID_FK int, ITEM_ID int, ITEM_NAME varchar(255), QUANTITY int);
